#include <iostream>
#include <string>
#include <vector>
#include <windows.h>
#include <tchar.h>

#include <oracl.h>
#include <obound.h>

#include "DBDialog.h"

void sendBonusToEmps(int amount, int emp)
{
	try
	{
		ODatabase db = OO4O_Test::DBDialog::LoginNewSession();
		db.GetSession().BeginTransaction();

		db.GetParameters().Add("amount", amount, OPARAMETER_INVAR, OTYPE_NUMBER);
		db.GetParameters().Add("emp", emp, OPARAMETER_INVAR, OTYPE_NUMBER);

		if( db.ExecuteSQL("begin HR_PACKAGE_1.send_bonus_emp(:amount, :emp); end;") != OSUCCESS )
		{
			std::cout << db.GetErrorText() << " " << db.GetServerErrorText() << std::endl;
			db.GetSession().Rollback();
			return;
		}

		db.GetSession().Commit();
	}
	catch (OException e)
	{
		
	}
}

int main()
{
	OStartup();
	
	sendBonusToEmps(10, 100);
	
	OShutdown();
	return 0;
}
